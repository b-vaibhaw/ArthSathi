def is_duplicate(text):
    m = MinHash(num_perm=128)
    for word in text.split(): m.update(word.encode('utf8'))
    result = lsh.query(m)
    if result: return True
    lsh.insert(hashlib.md5(text.encode()).hexdigest(), m)
    return False

4.1.1 Complete Model Code — arthasathi_model.py
# arthasathi_model.py — ArthaSathi Custom LLM from Scratch
# Architecture: Decoder-only Transformer (GPT-2 family)
# Enhancements: RoPE positional encoding, SwiGLU activation, RMSNorm
# These are LLaMA-style improvements over vanilla GPT-2
import torch
import torch.nn as nn
import torch.nn.functional as F
import math
from dataclasses import dataclass
@dataclass
class ArthaSathiConfig:
    vocab_size: int = 60000      # Custom multilingual vocab
    context_length: int = 1024   # Max tokens in context
    d_model: int = 1024          # Embedding dimension
    n_heads: int = 16            # Attention heads
    n_layers: int = 24           # Transformer blocks
    d_ff: int = 4096             # Feedforward hidden size
    dropout: float = 0.1
    rope_base: float = 10000.0   # RoPE frequency base
    # Total params with these settings: ~345M
class RMSNorm(nn.Module):  # Better than LayerNorm (LLaMA style)
    def __init__(self, dim, eps=1e-6):
        super().__init__()
        self.eps = eps
        self.weight = nn.Parameter(torch.ones(dim))
    def forward(self, x):
        return x * torch.rsqrt(x.pow(2).mean(-1, keepdim=True) + self.eps) * self.weight
class RotaryEmbedding(nn.Module):  # RoPE — better than learned positions
    def __init__(self, dim, base=10000):
        super().__init__()
        inv_freq = 1.0 / (base ** (torch.arange(0,dim,2).float() / dim))
        self.register_buffer('inv_freq', inv_freq)
    def forward(self, seq_len, device):
        t = torch.arange(seq_len, device=device).float()
        freqs = torch.einsum('i,j->ij', t, self.inv_freq)
        return torch.cat([freqs, freqs], dim=-1)
def rotate_half(x):
    x1, x2 = x[..., :x.shape[-1]//2], x[..., x.shape[-1]//2:]
    return torch.cat([-x2, x1], dim=-1)
def apply_rope(q, k, cos, sin):
    return (q * cos) + (rotate_half(q) * sin), (k * cos) + (rotate_half(k) * sin)
class SwiGLU(nn.Module):  # Better activation than ReLU/GELU
    def __init__(self, d_model, d_ff):
        super().__init__()
        self.w1 = nn.Linear(d_model, d_ff, bias=False)
        self.w2 = nn.Linear(d_ff, d_model, bias=False)
        self.w3 = nn.Linear(d_model, d_ff, bias=False)
    def forward(self, x):
        return self.w2(F.silu(self.w1(x)) * self.w3(x))
class MultiHeadAttention(nn.Module):
    def __init__(self, config):
        super().__init__()
        assert config.d_model % config.n_heads == 0
        self.n_heads = config.n_heads
        self.head_dim = config.d_model // config.n_heads
        self.Wq = nn.Linear(config.d_model, config.d_model, bias=False)
        self.Wk = nn.Linear(config.d_model, config.d_model, bias=False)
        self.Wv = nn.Linear(config.d_model, config.d_model, bias=False)
        self.Wo = nn.Linear(config.d_model, config.d_model, bias=False)
        self.rope = RotaryEmbedding(self.head_dim, config.rope_base)
        self.dropout = nn.Dropout(config.dropout)
    def forward(self, x, mask=None):
        B, T, C = x.shape
        q,k,v = self.Wq(x), self.Wk(x), self.Wv(x)
        q = q.view(B,T,self.n_heads,self.head_dim).transpose(1,2)
        k = k.view(B,T,self.n_heads,self.head_dim).transpose(1,2)
        v = v.view(B,T,self.n_heads,self.head_dim).transpose(1,2)
        cos_sin = self.rope(T, x.device)
        cos = cos_sin.cos()[None,None,:,:]
        sin = cos_sin.sin()[None,None,:,:]
        q, k = apply_rope(q, k, cos, sin)
        # Flash attention if available, else standard
        attn = torch.nn.functional.scaled_dot_product_attention(
            q, k, v, attn_mask=mask, dropout_p=self.dropout.p if self.training else 0.0,
            is_causal=True)
        out = attn.transpose(1,2).contiguous().view(B,T,C)
        return self.Wo(out)
class TransformerBlock(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.norm1 = RMSNorm(config.d_model)
        self.attn = MultiHeadAttention(config)
        self.norm2 = RMSNorm(config.d_model)
        self.ffn = SwiGLU(config.d_model, config.d_ff)
    def forward(self, x):
        x = x + self.attn(self.norm1(x))  # Pre-norm residual
        x = x + self.ffn(self.norm2(x))
        return x
class ArthaSathiLLM(nn.Module):
    def __init__(self, config):
        super().__init__()
        self.config = config
        self.embedding = nn.Embedding(config.vocab_size, config.d_model)
        self.blocks = nn.ModuleList([TransformerBlock(config) for _ in range(config.n_layers)])
        self.norm = RMSNorm(config.d_model)
        self.lm_head = nn.Linear(config.d_model, config.vocab_size, bias=False)
        # Weight tying: embedding and output share weights (saves params + improves quality)
        self.lm_head.weight = self.embedding.weight
        self.apply(self._init_weights)
    def _init_weights(self, module):
        if isinstance(module, nn.Linear):
            torch.nn.init.normal_(module.weight, mean=0.0, std=0.02)
            if module.bias is not None: torch.nn.init.zeros_(module.bias)
        elif isinstance(module, nn.Embedding):
            torch.nn.init.normal_(module.weight, mean=0.0, std=0.02)
    def forward(self, input_ids, targets=None):
        B, T = input_ids.shape
        x = self.embedding(input_ids)
        for block in self.blocks: x = block(x)
        x = self.norm(x)
        logits = self.lm_head(x)
        loss = None
        if targets is not None:
            loss = F.cross_entropy(logits.view(-1, self.config.vocab_size), targets.view(-1))
        return logits, loss
    def param_count(self):
        return sum(p.numel() for p in self.parameters())

scheduler = CosineAnnealingLR(optimizer, T_max=MAX_STEPS, eta_min=LEARNING_RATE*0.1)
scaler = amp.GradScaler()  # Mixed precision training (fp16)
# Training loop
model.train()
for step in range(MAX_STEPS):
    optimizer.zero_grad()
    loss_accum = 0.0
    for micro_step in range(GRAD_ACCUM):
        x, y = get_batch(train_data, BATCH_SIZE, CONTEXT_LEN, device)
        with amp.autocast(device_type='cuda', dtype=torch.float16):
            logits, loss = model(x, y)
            loss = loss / GRAD_ACCUM
        scaler.scale(loss).backward()
        loss_accum += loss.item()
    # Gradient clipping (prevents exploding gradients)
    scaler.unscale_(optimizer)
    torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
    scaler.step(optimizer)
    scaler.update()
    scheduler.step()
    if step % 100 == 0:
        print(f'Step {step} | Loss: {loss_accum:.4f} | LR: {scheduler.get_last_lr()[0]:.2e}')
    if step % SAVE_EVERY == 0:
        torch.save({'model': model.state_dict(), 'config': config,
                    'step': step, 'optimizer': optimizer.state_dict()},
                   f'checkpoints/arthasathi_step{step}.pt')

SECTION 5: RAG ENGINE — KNOWLEDGE + MEMORY
5.1 What RAG Does for ArthaSathi
RAG (Retrieval Augmented Generation) gives ArthaSathi access to up-to-date knowledge without retraining. When a user asks about current interest rates, specific GST rules, or a bank's hardship program, the RAG engine retrieves the relevant information from a vector database and injects it into the model's context. This prevents hallucination on factual financial queries — the single biggest risk in a financial AI product.
RAG Knowledge Base Contents
RBI circulars and interest rate policies. GST rules for informal sector (updated live). State-wise microfinance regulations. Bank-specific hardship program eligibility criteria. EMI calculator, debt avalanche/snowball formulas. Tax slabs, ITR filing rules for gig workers. Creditor negotiation scripts (template library).
RAG Memory Layer (Per User)
User's complete debt profile (loan amounts, rates, lenders). 90-day income and expense history (voice note transcriptions). Business transaction records for micro-entrepreneurs. Saved negotiation scripts and outcomes. Language preference and literacy level. Risk score and capacity-to-repay model.
5.2 RAG Implementation Code
# rag_engine.py — Complete RAG system for ArthaSathi
import faiss
import numpy as np
from sentence_transformers import SentenceTransformer
import json, sqlite3
# Use multilingual sentence encoder for 9 Indian languages
# 'paraphrase-multilingual-MiniLM-L12-v2' supports all our languages
encoder = SentenceTransformer('paraphrase-multilingual-MiniLM-L12-v2')
class ArthaSathiRAG:
    def __init__(self, index_path='faiss_index', db_path='arthakb.db'):
        # FAISS vector store for fast similarity search
        self.index = faiss.IndexFlatIP(384)  # 384-dim embeddings
        self.documents = []
        self.db = sqlite3.connect(db_path)
        self._init_db()
    def _init_db(self):
        self.db.execute('''CREATE TABLE IF NOT EXISTS user_memory (
            user_id TEXT, key TEXT, value TEXT, updated_at TIMESTAMP,
            PRIMARY KEY (user_id, key))''')
        self.db.execute('''CREATE TABLE IF NOT EXISTS transactions (
            user_id TEXT, type TEXT, amount REAL, description TEXT,
            language TEXT, timestamp TIMESTAMP)''')
        self.db.commit()
    def add_knowledge(self, text, metadata):
        embedding = encoder.encode([text])[0]
        embedding = embedding / np.linalg.norm(embedding)  # Normalize
        self.index.add(embedding.reshape(1, -1))
        self.documents.append({'text': text, 'metadata': metadata})
    def retrieve(self, query, k=5):
        query_emb = encoder.encode([query])[0]
        query_emb = query_emb / np.linalg.norm(query_emb)
        scores, indices = self.index.search(query_emb.reshape(1,-1), k)
        return [self.documents[i] for i in indices[0] if i < len(self.documents)]
    def save_user_memory(self, user_id, key, value):
        self.db.execute('INSERT OR REPLACE INTO user_memory VALUES (?,?,?,datetime("now"))',
                        (user_id, key, json.dumps(value)))
        self.db.commit()
    def get_user_profile(self, user_id):
        rows = self.db.execute('SELECT key, value FROM user_memory WHERE user_id=?',
                                (user_id,)).fetchall()
        return {k: json.loads(v) for k, v in rows}
    def build_context(self, user_id, user_message):
        # Retrieve relevant knowledge
        docs = self.retrieve(user_message, k=3)
        knowledge_context = '\n'.join([d['text'] for d in docs])
        # Get user's financial profile
        profile = self.get_user_profile(user_id)
        # Build full context string injected into LLM prompt
        return f'''[KNOWLEDGE]\n{knowledge_context}\n[USER PROFILE]\n{json.dumps(profile, ensure_ascii=False)}'''

    tts.write_to_fp(audio_buffer)
    return audio_buffer.getvalue()

# debt_engine.py
from dataclasses import dataclass
from typing import List
import json
@dataclass
class Debt:
    name: str         # 'HDFC Credit Card', 'Moneylender Ramesh'
    principal: float  # Outstanding amount
    interest_rate: float  # Annual percentage
    min_payment: float    # Minimum monthly payment
    lender_type: str  # 'bank', 'nbfc', 'moneylender', 'medical', 'payday'
class DebtEngine:
    def avalanche_order(self, debts: List[Debt]):
        """Pay minimum on all, throw extra money at highest interest first"""
        return sorted(debts, key=lambda d: d.interest_rate, reverse=True)
    def snowball_order(self, debts: List[Debt]):
        """Pay minimum on all, throw extra money at smallest balance first (psychological wins)"""
        return sorted(debts, key=lambda d: d.principal)
    def months_to_payoff(self, principal, annual_rate, monthly_payment):
        """How many months to pay off this debt at this payment amount"""
        if monthly_payment <= 0: return float('inf')
        monthly_rate = annual_rate / 12 / 100
        if monthly_rate == 0: return principal / monthly_payment
        import math
        n = -math.log(1 - (monthly_rate * principal / monthly_payment)) / math.log(1 + monthly_rate)
        return max(0, n)
    def total_interest_paid(self, principal, annual_rate, months):
        monthly_rate = annual_rate / 12 / 100
        total = principal * (1 + monthly_rate) ** months
        return total - principal
    def generate_negotiation_script(self, debt: Debt, user_income: float, language: str):
        """Generate actual script for user to call/message creditor"""
        hardship_ratio = debt.min_payment / user_income
        if hardship_ratio > 0.4:  # Paying >40% of income on one debt = genuine hardship
            situation = 'severe hardship'
        elif hardship_ratio > 0.25:
            situation = 'moderate hardship'
        else:
            situation = 'standard negotiation'
        # Return to LLM as context to generate language-specific script
        return {
            'debt': debt.name,
            'lender_type': debt.lender_type,
            'situation': situation,
            'suggested_asks': ['interest rate reduction', 'EMI restructuring', 'moratorium period'],
            'legal_options': self.get_legal_options(debt.lender_type),
            'generate_in_language': language
        }
    def get_legal_options(self, lender_type):
        options = {
            'bank': ['RBI Ombudsman complaint', 'DRT (Debt Recovery Tribunal)', 'IBC Section 8 notice'],
            'nbfc': ['RBI NBFC regulation complaint', 'consumer court'],
            'moneylender': ['State Moneylenders Act', 'police complaint for usurious rates'],
            'medical': ['hospital financial assistance program', 'state health scheme'],
            'payday': ['RBI digital lending guidelines', 'Fair Practices Code complaint']
        }
        return options.get(lender_type, ['consumer court', 'legal aid'])

# Knowledge distillation training
def distillation_loss(student_logits, teacher_logits, true_labels, temperature=4.0):
    soft_teacher = F.softmax(teacher_logits / temperature, dim=-1)
    soft_student = F.log_softmax(student_logits / temperature, dim=-1)
    kd_loss = F.kl_div(soft_student, soft_teacher, reduction='batchmean') * (temperature ** 2)
    ce_loss = F.cross_entropy(student_logits, true_labels)
    return 0.7 * kd_loss + 0.3 * ce_loss
